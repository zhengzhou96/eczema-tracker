Open `lib/content/tips.ts`. The file currently has `getTodaysTip`. Add below it:

```ts
function getDayOfYear(today: Date): number {
  const start = Date.UTC(today.getUTCFullYear(), 0, 0);
  const now = Date.UTC(
    today.getUTCFullYear(),
    today.getUTCMonth(),
    today.getUTCDate(),
  );
  return Math.floor((now - start) / 86_400_000);
}
```

Then replace the existing `getTodaysTip` body to use it:

```ts
export function getTodaysTip(today: Date = new Date()): Tip {
  return tips[getDayOfYear(today) % tips.length];
}

export function getTips(count: number, today: Date = new Date()): Tip[] {
  const base = getDayOfYear(today);
  return Array.from(
    { length: count },
    (_, i) => tips[(base + i) % tips.length],
  );
}
```

- [ ] **Step 1: Create the install prompt component**

Create `components/install-prompt.tsx`:

```tsx
"use client";

import { useEffect, useState } from "react";
import { Download, X } from "lucide-react";

interface BeforeInstallPromptEvent extends Event {
  prompt(): Promise<void>;
  readonly userChoice: Promise<{ outcome: "accepted" | "dismissed" }>;
}

export function InstallPrompt() {
  const [deferredPrompt, setDeferredPrompt] =
    useState<BeforeInstallPromptEvent | null>(null);
  const [isIos, setIsIos] = useState(false);
  const [dismissed, setDismissed] = useState(false);

  useEffect(() => {
    if (typeof window === "undefined") return;

    const alreadyInstalled = window.matchMedia(
      "(display-mode: standalone)",
    ).matches;
    if (alreadyInstalled) return;

    if (sessionStorage.getItem("install-prompt-dismissed")) {
      setDismissed(true);
      return;
    }

    const ua = navigator.userAgent;
    const isIosDevice =
      /iphone|ipad|ipod/i.test(ua) && !(window as Window & { MSStream?: unknown }).MSStream;
    if (isIosDevice) {
      setIsIos(true);
      return;
    }

    function handler(e: Event) {
      e.preventDefault();
      setDeferredPrompt(e as BeforeInstallPromptEvent);
    }

    window.addEventListener("beforeinstallprompt", handler);
    return () => window.removeEventListener("beforeinstallprompt", handler);
  }, []);

  function dismiss() {
    sessionStorage.setItem("install-prompt-dismissed", "1");
    setDeferredPrompt(null);
    setIsIos(false);
    setDismissed(true);
  }

  async function install() {
    if (!deferredPrompt) return;
    await deferredPrompt.prompt();
    const { outcome } = await deferredPrompt.userChoice;
    if (outcome === "accepted") {
      setDeferredPrompt(null);
    }
  }

  if (dismissed || (!deferredPrompt && !isIos)) return null;

  return (
    <div className="fixed inset-x-0 bottom-[calc(env(safe-area-inset-bottom)+72px)] z-50 mx-auto flex w-full max-w-[480px] items-start gap-3 rounded-3xl border border-border bg-card px-4 py-3 shadow-lg">
      <div className="flex size-10 shrink-0 items-center justify-center rounded-2xl bg-primary text-primary-foreground">
        <Download className="size-5" aria-hidden />
      </div>
      <div className="min-w-0 flex-1">
        {isIos ? (
          <>
            <p className="text-sm font-bold">Add to Home Screen</p>
            <p className="mt-0.5 text-xs font-medium text-muted-foreground">
              Tap the share icon then &ldquo;Add to Home Screen&rdquo; for the
              app experience.
            </p>
          </>
        ) : (
          <>
            <p className="text-sm font-bold">Install EczemaTrack</p>
            <p className="mt-0.5 text-xs font-medium text-muted-foreground">
              Add to your home screen for faster access.
            </p>
            <button
              type="button"
              onClick={install}
              className="mt-2 rounded-full bg-primary px-4 py-1.5 text-xs font-bold text-primary-foreground transition-transform hover:scale-[1.02] active:scale-95"
            >
              Install
            </button>
          </>
        )}
      </div>
      <button
        type="button"
        onClick={dismiss}
        aria-label="Dismiss install prompt"
        className="flex size-8 shrink-0 items-center justify-center rounded-full bg-muted text-muted-foreground transition-transform hover:scale-105 hover:text-foreground active:scale-95"
      >
        <X className="size-4" aria-hidden />
      </button>
    </div>
  );
}
```
